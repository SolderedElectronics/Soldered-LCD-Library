25/* This struct helps us constructing the I2C output based on data and control outputs.
26 Because the LCD is set to 4-bit mode, 4 bits of the I2C output are for the control outputs
27 while the other 4 bits are for the 8 bits of data which are send in parts using the enable output.*/
28struct OutputState
29{
30 uint8_t rs = 0;
31 uint8_t rw = 0;
32 uint8_t E = 0;
33 uint8_t Led = 0;
34 uint8_t data = 0;
35
36 uint8_t GetLowData()
37 {
38 uint8_t buffer = rs;
39 buffer |= rw << 1;
40 buffer |= E << 2;
41 buffer |= Led << 3;
42 buffer |= (data & 0x0F) << 4;
43
44 return buffer;
45 }
46
47 uint8_t GetHighData()
48 {
49 uint8_t buffer = rs;
50 buffer |= rw << 1;
51 buffer |= E << 2;
52 buffer |= Led << 3;
53 buffer |= (data & 0xF0);
54 return buffer;
55 }
56};
